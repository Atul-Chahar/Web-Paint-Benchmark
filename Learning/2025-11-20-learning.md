const startAudit = () => {
    const urlInput = document.getElementById('target-url').value;
    
    if (!urlInput) {
        alert("Please enter a valid URL");
        return;
    }

    // 1. Hide the Input Interface
    document.querySelector('.landing-page').style.display = 'none';
    
    // 2. Show the Loading/Processing State
    document.querySelector('.loader').style.display = 'block';

    // 3. Initialize the Benchmark Engine (Next Step)
    console.log(`Starting audit for: ${urlInput}`);
    // initPerformanceObserver(); 
};

document.getElementById('generate-report-btn').addEventListener('click', startAudit);
